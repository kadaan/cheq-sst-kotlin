import ai.cheq.sst.android.core.Config

fun main() { 
   //sampleStart

    Config("di_demo", "t.nc0.co", "sst", "nexus.ensighten.com", "digitalData", true)

println(event.publishPath) // sst

println(event.dataLayerName) // digitalData
println(event.debug) // true 
   //sampleEnd
}

val nexusHost: String

The alternative Ensighten domain for loading tags from a source other than nexus.ensighten.com within the SST instance. Typically only used with a value of “nexus-test.ensighten.com” when testing tags in a committed state.
