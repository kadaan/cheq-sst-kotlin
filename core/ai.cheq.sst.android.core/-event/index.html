Event

data class Event @JvmOverloads constructor(val name: String, val data: Map<String, Any> = emptyMap(), val parameters: Map<String, String> = emptyMap())

Represents an event that can be sent to CHEQ SST. The event must have a name, and can optionally include data and parameters.

Links

More information about parameters that can be passed to the CHEQ SST service, see Getting Started with Data Collection Pixels

Samples

import ai.cheq.sst.android.core.Event

fun main() { 
   //sampleStart 
   // Create an event with just a name
val event = Event("screen_view")
println(event.name) // screen_view 
   //sampleEnd
}
import ai.cheq.sst.android.core.Event

fun main() { 
   //sampleStart 
   // Create an event with a name and data
val event = Event("screen_view", mapOf("screen_name" to "Home"))
println(event.name) // screen_view
println(event.data["screen_name"]) // Home 
   //sampleEnd
}
import ai.cheq.sst.android.core.Event

fun main() { 
   //sampleStart 
   // Create an event with a name, data, and parameters
val event = Event(
    "screen_view", mapOf("screen_name" to "Home"), mapOf("ensDisableTracking" to "user")
)
println(event.name) // screen_view
println(event.data["screen_name"]) // Home
println(event.parameters["ensDisableTracking"]) // user 
   //sampleEnd
}
