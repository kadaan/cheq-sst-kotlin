configure

@JvmStatic
fun configure(config: Config, contextProvider: ContextProvider)

Configures the CHEQ SST SDK with the specified contextProvider and config.

This must be called before any other methods in the SDK are used.

Parameters

config

The Config object to use for the SDK.

contextProvider

The Android ContextProvider to use for the SDK.

import ai.cheq.sst.android.core.Config
import ai.cheq.sst.android.core.Event
import ai.cheq.sst.android.core.Sst
import ai.cheq.sst.android.core.models.Model
import ai.cheq.sst.android.core.models.ModelContext
import ai.cheq.sst.android.core.models.Models

fun main() { 
   //sampleStart 
   runBlocking {
            // Create a custom model
            class CustomModel : Model<CustomModel.Data>(Data::class, "custom_model") {
                override suspend fun get(modelContext: ModelContext): Data {
                    return Data()
                }

                inner class Data : Model.Data() {
                    @JsonProperty
                    val foo: String = "bar"
                }
            }

            // Configure the SDK with a custom model
            Sst.configure(
                Config(
                    clientName = "client_name", models = Models(CustomModel())
                )
            ) { context }

            with(Sst.config()) {
                println(clientName) // client_name
                println(models.all().size) // 4

                val customModel = models.get<CustomModel>()!!
                println(customModel.identifier.key) // custom_model
            }
        } 
   //sampleEnd
}
